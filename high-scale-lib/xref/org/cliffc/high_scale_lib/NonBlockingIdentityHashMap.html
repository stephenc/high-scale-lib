15  /**
16   * A lock-free alternate implementation of {@link java.util.concurrent.ConcurrentHashMap}
17   * with better scaling properties and generally lower costs to mutate the Map.
18   * It provides identical correctness properties as ConcurrentHashMap.  All
19   * operations are non-blocking and multi-thread safe, including all update
20   * operations.  {@link NonBlockingHashMap} scales substatially better than
21   * {@link java.util.concurrent.ConcurrentHashMap} for high update rates, even with a
22   * large concurrency factor.  Scaling is linear up to 768 CPUs on a 768-CPU
23   * Azul box, even with 100% updates or 100% reads or any fraction in-between.
24   * Linear scaling up to all cpus has been observed on a 32-way Sun US2 box,
25   * 32-way Sun Niagra box, 8-way Intel box and a 4-way Power box.
26   *
27   * This class obeys the same functional specification as {@link
28   * java.util.Hashtable}, and includes versions of methods corresponding to
29   * each method of <tt>Hashtable</tt>. However, even though all operations are
30   * thread-safe, operations do <em>not</em> entail locking and there is
31   * <em>not</em> any support for locking the entire table in a way that
32   * prevents all access.  This class is fully interoperable with
33   * <tt>Hashtable</tt> in programs that rely on its thread safety but not on
34   * its synchronization details.
35   *
36   * <p> Operations (including <tt>put</tt>) generally do not block, so may
37   * overlap with other update operations (including other <tt>puts</tt> and
38   * <tt>removes</tt>).  Retrievals reflect the results of the most recently
39   * <em>completed</em> update operations holding upon their onset.  For
40   * aggregate operations such as <tt>putAll</tt>, concurrent retrievals may
41   * reflect insertion or removal of only some entries.  Similarly, Iterators
42   * and Enumerations return elements reflecting the state of the hash table at
43   * some point at or since the creation of the iterator/enumeration.  They do
44   * <em>not</em> throw {@link ConcurrentModificationException}.  However,
45   * iterators are designed to be used by only one thread at a time.
46   *
47   * <p> Very full tables, or tables with high reprobe rates may trigger an
48   * internal resize operation to move into a larger table.  Resizing is not
49   * terribly expensive, but it is not free either; during resize operations
50   * table throughput may drop somewhat.  All threads that visit the table
51   * during a resize will 'help' the resizing but will still be allowed to
52   * complete their operation before the resize is finished (i.e., a simple
53   * 'get' operation on a million-entry table undergoing resizing will not need
54   * to block until the entire million entries are copied).
55   *
56   * <p>This class and its views and iterators implement all of the
57   * <em>optional</em> methods of the {@link Map} and {@link Iterator}
58   * interfaces.
59   *
60   * <p> Like {@link Hashtable} but unlike {@link HashMap}, this class
61   * does <em>not</em> allow <tt>null</tt> to be used as a key or value.
62   *
63   *
64   * @since 1.5
65   * @author Cliff Click
66   * @param <TypeK> the type of keys maintained by this map
67   * @param <TypeV> the type of mapped values
68   * 
69   *  @author Prashant Deva 
70   *  Modified from original NonBlockingHashMap to use identity equality.
71   *  Uses System.identityHashCode() to calculate hashMap.
72   *  Key equality is compared using '=='.
73   */
74

76    extends AbstractMap<TypeK, TypeV>
77    implements ConcurrentMap<TypeK, TypeV>, Cloneable, Serializable {

83    // --- Bits to allow Unsafe access to arrays
84    private static final Unsafe _unsafe = UtilUnsafe.getUnsafe();
85    private static final int _Obase  = _unsafe.arrayBaseOffset(Object[].class);
86    private static final int _Oscale = _unsafe.arrayIndexScale(Object[].class);
87    private static long rawIndex(final Object[] ary, final int idx) {
88      assert idx >= 0 && idx < ary.length;
89      return _Obase + idx * _Oscale;
90    }

92    // --- Setup to use Unsafe
93    private static final long _kvs_offset;
94    static {                      // <clinit>
95      Field f = null;
96      try { f = NonBlockingHashMap.class.getDeclaredField("_kvs"); }
97      catch( java.lang.NoSuchFieldException e ) { throw new RuntimeException(e); }
98      _kvs_offset = _unsafe.objectFieldOffset(f);
99    }
100   private final boolean CAS_kvs( final Object[] oldkvs, final Object[] newkvs ) {
101     return _unsafe.compareAndSwapObject(this, _kvs_offset, oldkvs, newkvs );
102   }

104   // --- Adding a 'prime' bit onto Values via wrapping with a junk wrapper class
105   private static final class Prime {
106     final Object _V;
107     Prime( Object V ) { _V = V; }
108     static Object unbox( Object V ) { return V instanceof Prime ? ((Prime)V)._V : V; }
109   }

111   // --- hash ----------------------------------------------------------------
112   // Helper function to spread lousy hashCodes
113   private static final int hash(final Object key) {
114     int h = System.identityHashCode(key); // The real hashCode call
115     h ^= (h>>>20) ^ (h>>>12);
116     h ^= (h>>> 7) ^ (h>>> 4);
117     return h;
118   }

120   // --- The Hash Table --------------------
121   // Slot 0 is always used for a 'CHM' entry below to hold the interesting
122   // bits of the hash table.  Slot 1 holds full hashes as an array of ints.
123   // Slots {2,3}, {4,5}, etc hold {Key,Value} pairs.  The entire hash table
124   // can be atomically replaced by CASing the _kvs field.
125   //
126   // Why is CHM buried inside the _kvs Object array, instead of the other way
127   // around?  The CHM info is used during resize events and updates, but not
128   // during standard 'get' operations.  I assume 'get' is much more frequent
129   // than 'put'.  'get' can skip the extra indirection of skipping through the
130   // CHM to reach the _kvs array.
131   private transient Object[] _kvs;
132   private static final CHM   chm   (Object[] kvs) { return (CHM  )kvs[0]; }
133   private static final int[] hashes(Object[] kvs) { return (int[])kvs[1]; }
134   // Number of K,V pairs in the table
135   private static final int len(Object[] kvs) { return (kvs.length-2)>>1; }
136 
137   // Time since last resize
138   private transient long _last_resize_milli;

140   // --- Minimum table size ----------------
141   // Pick size 8 K/V pairs, which turns into (8*2+2)*4+12 = 84 bytes on a
142   // standard 32-bit HotSpot, and (8*2+2)*8+12 = 156 bytes on 64-bit Azul.
143   private static final int MIN_SIZE_LOG=3;             //
144   private static final int MIN_SIZE=(1<<MIN_SIZE_LOG); // Must be power of 2
145 
146   // --- Sentinels -------------------------
147   // No-Match-Old - putIfMatch does updates only if it matches the old value,
148   // and NO_MATCH_OLD basically counts as a wildcard match.
149   private static final Object NO_MATCH_OLD = new Object(); // Sentinel
150   // Match-Any-not-null - putIfMatch does updates only if it find a real old
151   // value.
152   private static final Object MATCH_ANY = new Object(); // Sentinel
153   // This K/V pair has been deleted (but the Key slot is forever claimed).
154   // The same Key can be reinserted with a new value later.
155   private static final Object TOMBSTONE = new Object();
156   // Prime'd or box'd version of TOMBSTONE.  This K/V pair was deleted, then a
157   // table resize started.  The K/V pair has been marked so that no new
158   // updates can happen to the old table (and since the K/V pair was deleted
159   // nothing was copied to the new table).
160   private static final Prime TOMBPRIME = new Prime(TOMBSTONE);
161 
162   // --- key,val -------------------------------------------------------------
163   // Access K,V for a given idx
164   //
165   // Note that these are static, so that the caller is forced to read the _kvs
166   // field only once, and share that read across all key/val calls - lest the
167   // _kvs field move out from under us and back-to-back key & val calls refer
168   // to different _kvs arrays.
169   private static final Object key(Object[] kvs,int idx) { return kvs[(idx<<1)+2]; }
170   private static final Object val(Object[] kvs,int idx) { return kvs[(idx<<1)+3]; }
171   private static final boolean CAS_key( Object[] kvs, int idx, Object old, Object key ) {
172     return _unsafe.compareAndSwapObject( kvs, rawIndex(kvs,(idx<<1)+2), old, key );
173   }
174   private static final boolean CAS_val( Object[] kvs, int idx, Object old, Object val ) {
175     return _unsafe.compareAndSwapObject( kvs, rawIndex(kvs,(idx<<1)+3), old, val );
176   }

179   // --- dump ----------------------------------------------------------------
180   /** Verbose printout of table internals, useful for debugging.  */
181   public final void print() {
182     System.out.println("=========");
183     print2(_kvs);
184     System.out.println("=========");
185   }
186   // print the entire state of the table
187   private final void print( Object[] kvs ) {
188     for( int i=0; i<len(kvs); i++ ) {
189       Object K = key(kvs,i);
190       if( K != null ) {
191         String KS = (K == TOMBSTONE) ? "XXX" : K.toString();
192         Object V = val(kvs,i);
193         Object U = Prime.unbox(V);
194         String p = (V==U) ? "" : "prime_";
195         String US = (U == TOMBSTONE) ? "tombstone" : U.toString();
196         System.out.println(""+i+" ("+KS+","+p+US+")");
197       }
198     }
199     Object[] newkvs = chm(kvs)._newkvs; // New table, if any
200     if( newkvs != null ) {
201       System.out.println("----");
202       print(newkvs);
203     }
204   }
205   // print only the live values, broken down by the table they are in
206   private final void print2( Object[] kvs) {
207     for( int i=0; i<len(kvs); i++ ) {
208       Object key = key(kvs,i);
209       Object val = val(kvs,i);
210       Object U = Prime.unbox(val);
211       if( key != null && key != TOMBSTONE &&  // key is sane
212           val != null && U   != TOMBSTONE ) { // val is sane
213         String p = (val==U) ? "" : "prime_";
214         System.out.println(""+i+" ("+key+","+p+val+")");
215       }
216     }
217     Object[] newkvs = chm(kvs)._newkvs; // New table, if any
218     if( newkvs != null ) {
219       System.out.println("----");
220       print2(newkvs);
221     }
222   }

226   /** Get and clear the current count of reprobes.  Reprobes happen on key
227    *  collisions, and a high reprobe rate may indicate a poor hash function or
228    *  weaknesses in the table resizing function.
229    *  @return the count of reprobes since the last call to {@link #reprobes}

234   // --- reprobe_limit -----------------------------------------------------
235   // Heuristic to decide if we have reprobed toooo many times.  Running over
236   // the reprobe limit on a 'get' call acts as a 'miss'; on a 'put' call it
237   // can trigger a table resize.  Several places must have exact agreement on
238   // what the reprobe_limit is, so we share it here.
239   private static final int reprobe_limit( int len ) {
240     return REPROBE_LIMIT + (len>>2);
241   }

246   /** Create a new NonBlockingHashMap with default minimum size (currently set
247    *  to 8 K/V pairs or roughly 84 bytes on a standard 32-bit JVM). */
248   public NonBlockingIdentityHashMap( ) { this(MIN_SIZE); }

250   /** Create a new NonBlockingHashMap with initial room for the given number of
251    *  elements, thus avoiding internal resizing operations to reach an
252    *  appropriate size.  Large numbers here when used with a small count of
253    *  elements will sacrifice space for a small amount of time gained.  The
254    *  initial size will be rounded up internally to the next larger power of 2. */
255   public NonBlockingIdentityHashMap( final int initial_sz ) { initialize(initial_sz); }
256   private final void initialize( int initial_sz ) {
257     if( initial_sz < 0 ) throw new IllegalArgumentException();
258     int i;                      // Convert to next largest power-of-2
259     if( initial_sz > 1024*1024 ) initial_sz = 1024*1024;
260     for( i=MIN_SIZE_LOG; (1<<i) < (initial_sz<<2); i++ ) ;
261     // Double size for K,V pairs, add 1 for CHM and 1 for hashes
262     _kvs = new Object[((1<<i)<<1)+2];
263     _kvs[0] = new CHM(new Counter()); // CHM in slot 0
264     _kvs[1] = new int[1<<i];          // Matching hash entries
265     _last_resize_milli = System.currentTimeMillis();
266   }
267   // Version for subclassed readObject calls, to be called after the defaultReadObject
268   protected final void initialize() { initialize(MIN_SIZE); }

342   private final TypeV putIfMatch( Object key, Object newVal, Object oldVal ) {
343     if (oldVal == null || newVal == null) throw new NullPointerException();
344     final Object res = putIfMatch( this, _kvs, key, newVal, oldVal );
345     assert !(res instanceof Prime);
346     assert res != null;
347     return res == TOMBSTONE ? null : (TypeV)res;
348   }

389   /**
390    * Creates a shallow copy of this hashtable. All the structure of the
391    * hashtable itself is copied, but the keys and values are not cloned.
392    * This is a relatively expensive operation.
393    *
394    * @return  a clone of the hashtable.
395    */
396   @Override
397   public Object clone() {
398     try {
399       // Must clone, to get the class right; NBHM might have been
400       // extended so it would be wrong to just make a new NBHM.
401       NonBlockingIdentityHashMap<TypeK,TypeV> t = (NonBlockingIdentityHashMap<TypeK,TypeV>) super.clone();
402       // But I don't have an atomic clone operation - the underlying _kvs
403       // structure is undergoing rapid change.  If I just clone the _kvs
404       // field, the CHM in _kvs[0] won't be in sync.
405       //
406       // Wipe out the cloned array (it was shallow anyways).
407       t.clear();
408       // Now copy sanely
409       for( TypeK K : keySet() ) {
410         final TypeV V = get(K);  // Do an official 'get'
411         t.put(K,V);
412       }
413       return t;
414     } catch (CloneNotSupportedException e) {
415       // this shouldn't happen, since we are Cloneable
416       throw new InternalError();
417     }
418   }

420   /**
421    * Returns a string representation of this map.  The string representation
422    * consists of a list of key-value mappings in the order returned by the
423    * map's <tt>entrySet</tt> view's iterator, enclosed in braces
424    * (<tt>"{}"</tt>).  Adjacent mappings are separated by the characters
425    * <tt>", "</tt> (comma and space).  Each key-value mapping is rendered as
426    * the key followed by an equals sign (<tt>"="</tt>) followed by the
427    * associated value.  Keys and values are converted to strings as by
428    * {@link String#valueOf(Object)}.
429    *
430    * @return a string representation of this map
431    */
432   @Override
433   public String toString() {
434     Iterator<Entry<TypeK,TypeV>> i = entrySet().iterator();
435     if( !i.hasNext())
436       return "{}";
437 
438     StringBuilder sb = new StringBuilder();
439     sb.append('{');
440     for (;;) {
441       Entry<TypeK,TypeV> e = i.next();
442       TypeK key = e.getKey();
443       TypeV value = e.getValue();
444       sb.append(key   == this ? "(this Map)" : key);
445       sb.append('=');
446       sb.append(value == this ? "(this Map)" : value);
447       if( !i.hasNext())
448         return sb.append('}').toString();
449       sb.append(", ");
450     }
451   }

454   /** Returns the value to which the specified key is mapped, or {@code null}
455    *  if this map contains no mapping for the key.
456    *  <p>More formally, if this map contains a mapping from a key {@code k} to
457    *  a value {@code v} such that {@code key.equals(k)}, then this method
458    *  returns {@code v}; otherwise it returns {@code null}.  (There can be at
459    *  most one such mapping.)
460    * @throws NullPointerException if the specified key is null */
461   // Never returns a Prime nor a Tombstone.
462   @Override
463   public TypeV get( Object key ) {
464     final int fullhash= hash (key); // throws NullPointerException if key is null
465     final Object V = get_impl(this,_kvs,key,fullhash);
466     assert !(V instanceof Prime); // Never return a Prime
467     return (TypeV)V;
468   }
469 
470   private static final Object get_impl( final NonBlockingIdentityHashMap topmap, final Object[] kvs, final Object key, final int fullhash ) {
471     final int len     = len  (kvs); // Count of key/value pairs, reads kvs.length
472     final CHM chm     = chm  (kvs); // The CHM, for a volatile read below; reads slot 0 of kvs
473 
474     int idx = fullhash & (len-1); // First key hash
475 
476     // Main spin/reprobe loop, looking for a Key hit
477     int reprobe_cnt=0;
478     while( true ) {
479       // Probe table.  Each read of 'val' probably misses in cache in a big
480       // table; hopefully the read of 'key' then hits in cache.
481       final Object K = key(kvs,idx); // Get key   before volatile read, could be null
482       final Object V = val(kvs,idx); // Get value before volatile read, could be null or Tombstone or Prime
483       if( K == null ) return null;   // A clear miss
484 
485       // We need a volatile-read here to preserve happens-before semantics on
486       // newly inserted Keys.  If the Key body was written just before inserting
487       // into the table a Key-compare here might read the uninitalized Key body.
488       // Annoyingly this means we have to volatile-read before EACH key compare.
489       // .
490       // We also need a volatile-read between reading a newly inserted Value
491       // and returning the Value (so the user might end up reading the stale
492       // Value contents).  Same problem as with keys - and the one volatile
493       // read covers both.
494       final Object[] newkvs = chm._newkvs; // VOLATILE READ before key compare
495 
496       // Key-compare
497       if( K == key ) {
498         // Key hit!  Check for no table-copy-in-progress
499         if( !(V instanceof Prime) ) // No copy?
500           return (V == TOMBSTONE) ? null : V; // Return the value
501         // Key hit - but slot is (possibly partially) copied to the new table.
502         // Finish the copy & retry in the new table.
503         return get_impl(topmap,chm.copy_slot_and_check(topmap,kvs,idx,key),key,fullhash); // Retry in the new table
504       }
505       // get and put must have the same key lookup logic!  But only 'put'
506       // needs to force a table-resize for a too-long key-reprobe sequence.
507       // Check for too-many-reprobes on get - and flip to the new table.
508       if( ++reprobe_cnt >= reprobe_limit(len) || // too many probes
509           key == TOMBSTONE ) // found a TOMBSTONE key, means no more keys in this table
510         return newkvs == null ? null : get_impl(topmap,topmap.help_copy(newkvs),key,fullhash); // Retry in the new table
511 
512       idx = (idx+1)&(len-1);    // Reprobe by 1!  (could now prefetch)
513     }
514   }
515 
516   // --- putIfMatch ---------------------------------------------------------
517   // Put, Remove, PutIfAbsent, etc.  Return the old value.  If the returned
518   // value is equal to expVal (or expVal is NO_MATCH_OLD) then the put can be
519   // assumed to work (although might have been immediately overwritten).  Only
520   // the path through copy_slot passes in an expected value of null, and
521   // putIfMatch only returns a null if passed in an expected null.
522   private static final Object putIfMatch( final NonBlockingIdentityHashMap topmap, final Object[] kvs, final Object key, final Object putval, final Object expVal ) {
523     assert putval != null;
524     assert !(putval instanceof Prime);
525     assert !(expVal instanceof Prime);
526     final int fullhash = hash  (key); // throws NullPointerException if key null
527     final int len      = len   (kvs); // Count of key/value pairs, reads kvs.length
528     final CHM chm      = chm   (kvs); // Reads kvs[0]
529     int idx = fullhash & (len-1);
530 
531     // ---
532     // Key-Claim stanza: spin till we can claim a Key (or force a resizing).
533     int reprobe_cnt=0;
534     Object K=null, V=null;
535     Object[] newkvs=null;
536     while( true ) {             // Spin till we get a Key slot
537       V = val(kvs,idx);         // Get old value (before volatile read below!)
538       K = key(kvs,idx);         // Get current key
539       if( K == null ) {         // Slot is free?
540         // Found an empty Key slot - which means this Key has never been in
541         // this table.  No need to put a Tombstone - the Key is not here!
542         if( putval == TOMBSTONE ) return putval; // Not-now & never-been in this table
543         // Claim the null key-slot
544         if( CAS_key(kvs,idx, null, key ) ) { // Claim slot for Key
545           chm._slots.add(1);      // Raise key-slots-used count
546           break;                  // Got it!
547         }
548         // CAS to claim the key-slot failed.
549         //
550         // This re-read of the Key points out an annoying short-coming of Java
551         // CAS.  Most hardware CAS's report back the existing value - so that
552         // if you fail you have a *witness* - the value which caused the CAS
553         // to fail.  The Java API turns this into a boolean destroying the
554         // witness.  Re-reading does not recover the witness because another
555         // thread can write over the memory after the CAS.  Hence we can be in
556         // the unfortunate situation of having a CAS fail *for cause* but
557         // having that cause removed by a later store.  This turns a
558         // non-spurious-failure CAS (such as Azul has) into one that can
559         // apparently spuriously fail - and we avoid apparent spurious failure
560         // by not allowing Keys to ever change.
561         K = key(kvs,idx);       // CAS failed, get updated value
562         assert K != null;       // If keys[idx] is null, CAS shoulda worked
563       }
564       // Key slot was not null, there exists a Key here
565 
566       // We need a volatile-read here to preserve happens-before semantics on
567       // newly inserted Keys.  If the Key body was written just before inserting
568       // into the table a Key-compare here might read the uninitalized Key body.
569       // Annoyingly this means we have to volatile-read before EACH key compare.
570       newkvs = chm._newkvs;     // VOLATILE READ before key compare
571 
572       if( K == key )
573         break;                  // Got it!
574 
575       // get and put must have the same key lookup logic!  Lest 'get' give
576       // up looking too soon.
577       //topmap._reprobes.add(1);
578       if( ++reprobe_cnt >= reprobe_limit(len) || // too many probes or
579           key == TOMBSTONE ) { // found a TOMBSTONE key, means no more keys
580         // We simply must have a new table to do a 'put'.  At this point a
581         // 'get' will also go to the new table (if any).  We do not need
582         // to claim a key slot (indeed, we cannot find a free one to claim!).
583         newkvs = chm.resize(topmap,kvs);
584         if( expVal != null ) topmap.help_copy(newkvs); // help along an existing copy
585         return putIfMatch(topmap,newkvs,key,putval,expVal);
586       }
587 
588       idx = (idx+1)&(len-1); // Reprobe!
589     } // End of spinning till we get a Key slot
590 
591     // ---
592     // Found the proper Key slot, now update the matching Value slot.  We
593     // never put a null, so Value slots monotonically move from null to
594     // not-null (deleted Values use Tombstone).  Thus if 'V' is null we
595     // fail this fast cutout and fall into the check for table-full.
596     if( putval == V ) return V; // Fast cutout for no-change
597 
598     // See if we want to move to a new table (to avoid high average re-probe
599     // counts).  We only check on the initial set of a Value from null to
600     // not-null (i.e., once per key-insert).  Of course we got a 'free' check
601     // of newkvs once per key-compare (not really free, but paid-for by the
602     // time we get here).
603     if( newkvs == null &&       // New table-copy already spotted?
604         // Once per fresh key-insert check the hard way
605         ((V == null && chm.tableFull(reprobe_cnt,len)) ||
606          // Or we found a Prime, but the JMM allowed reordering such that we
607          // did not spot the new table (very rare race here: the writing
608          // thread did a CAS of _newkvs then a store of a Prime.  This thread
609          // reads the Prime, then reads _newkvs - but the read of Prime was so
610          // delayed (or the read of _newkvs was so accelerated) that they
611          // swapped and we still read a null _newkvs.  The resize call below
612          // will do a CAS on _newkvs forcing the read.
613          V instanceof Prime) )
614       newkvs = chm.resize(topmap,kvs); // Force the new table copy to start
615     // See if we are moving to a new table.
616     // If so, copy our slot and retry in the new table.
617     if( newkvs != null )
618       return putIfMatch(topmap,chm.copy_slot_and_check(topmap,kvs,idx,expVal),key,putval,expVal);
619 
620     // ---
621     // We are finally prepared to update the existing table
622     while( true ) {
623       assert !(V instanceof Prime);
624 
625       // Must match old, and we do not?  Then bail out now.  Note that either V
626       // or expVal might be TOMBSTONE.  Also V can be null, if we've never
627       // inserted a value before.  expVal can be null if we are called from
628       // copy_slot.
629 
630       if( expVal != NO_MATCH_OLD && // Do we care about expected-Value at all?
631           V != expVal &&            // No instant match already?
632           (expVal != MATCH_ANY || V == TOMBSTONE || V == null) &&
633           !(V==null && expVal == TOMBSTONE) &&    // Match on null/TOMBSTONE combo
634           (expVal == null || !expVal.equals(V)) ) // Expensive equals check at the last
635         return V;                                 // Do not update!
636 
637       // Actually change the Value in the Key,Value pair
638       if( CAS_val(kvs, idx, V, putval ) ) {
639         // CAS succeeded - we did the update!
640         // Both normal put's and table-copy calls putIfMatch, but table-copy
641         // does not (effectively) increase the number of live k/v pairs.
642         if( expVal != null ) {
643           // Adjust sizes - a striped counter
644           if(  (V == null || V == TOMBSTONE) && putval != TOMBSTONE ) chm._size.add( 1);
645           if( !(V == null || V == TOMBSTONE) && putval == TOMBSTONE ) chm._size.add(-1);
646         }
647         return (V==null && expVal!=null) ? TOMBSTONE : V;
648       } 
649       // Else CAS failed
650       V = val(kvs,idx);         // Get new value
651       // If a Prime'd value got installed, we need to re-run the put on the
652       // new table.  Otherwise we lost the CAS to another racing put.
653       // Simply retry from the start.
654       if( V instanceof Prime )
655         return putIfMatch(topmap,chm.copy_slot_and_check(topmap,kvs,idx,expVal),key,putval,expVal);
656     }
657   }
658 
659   // --- help_copy ---------------------------------------------------------
660   // Help along an existing resize operation.  This is just a fast cut-out
661   // wrapper, to encourage inlining for the fast no-copy-in-progress case.  We
662   // always help the top-most table copy, even if there are nested table
663   // copies in progress.
664   private final Object[] help_copy( Object[] helper ) {
665     // Read the top-level KVS only once.  We'll try to help this copy along,
666     // even if it gets promoted out from under us (i.e., the copy completes
667     // and another KVS becomes the top-level copy).
668     Object[] topkvs = _kvs;
669     CHM topchm = chm(topkvs);
670     if( topchm._newkvs == null ) return helper; // No copy in-progress
671     topchm.help_copy_impl(this,topkvs,false);
672     return helper;
673   }

676   // --- CHM -----------------------------------------------------------------
677   // The control structure for the NonBlockingIdentityHashMap
678   private static final class CHM<TypeK,TypeV> {
679     // Size in active K,V pairs
680     private final Counter _size;
681     public int size () { return (int)_size.get(); }
682 
683     // ---
684     // These next 2 fields are used in the resizing heuristics, to judge when
685     // it is time to resize or copy the table.  Slots is a count of used-up
686     // key slots, and when it nears a large fraction of the table we probably
687     // end up reprobing too much.  Last-resize-milli is the time since the
688     // last resize; if we are running back-to-back resizes without growing
689     // (because there are only a few live keys but many slots full of dead
690     // keys) then we need a larger table to cut down on the churn.
691 
692     // Count of used slots, to tell when table is full of dead unusable slots
693     private final Counter _slots;
694     public int slots() { return (int)_slots.get(); }
695 
696     // ---
697     // New mappings, used during resizing.
698     // The 'new KVs' array - created during a resize operation.  This
699     // represents the new table being copied from the old one.  It's the
700     // volatile variable that is read as we cross from one table to the next,
701     // to get the required memory orderings.  It monotonically transits from
702     // null to set (once).
703     volatile Object[] _newkvs;
704     private final AtomicReferenceFieldUpdater<CHM,Object[]> _newkvsUpdater =
705       AtomicReferenceFieldUpdater.newUpdater(CHM.class,Object[].class, "_newkvs");
706     // Set the _next field if we can.
707     boolean CAS_newkvs( Object[] newkvs ) {
708       while( _newkvs == null )
709         if( _newkvsUpdater.compareAndSet(this,null,newkvs) )
710           return true;
711       return false;
712     }
713     // Sometimes many threads race to create a new very large table.  Only 1
714     // wins the race, but the losers all allocate a junk large table with
715     // hefty allocation costs.  Attempt to control the overkill here by
716     // throttling attempts to create a new table.  I cannot really block here
717     // (lest I lose the non-blocking property) but late-arriving threads can
718     // give the initial resizing thread a little time to allocate the initial
719     // new table.  The Right Long Term Fix here is to use array-lets and
720     // incrementally create the new very large array.  In C I'd make the array
721     // with malloc (which would mmap under the hood) which would only eat
722     // virtual-address and not real memory - and after Somebody wins then we
723     // could in parallel initialize the array.  Java does not allow
724     // un-initialized array creation (especially of ref arrays!).
725     volatile long _resizers; // count of threads attempting an initial resize
726     private static final AtomicLongFieldUpdater<CHM> _resizerUpdater =
727       AtomicLongFieldUpdater.newUpdater(CHM.class, "_resizers");
728 
729     // ---
730     // Simple constructor
731     CHM( Counter size ) {
732       _size = size;
733       _slots= new Counter();
734     }
735 
736     // --- tableFull ---------------------------------------------------------
737     // Heuristic to decide if this table is too full, and we should start a
738     // new table.  Note that if a 'get' call has reprobed too many times and
739     // decided the table must be full, then always the estimate_sum must be
740     // high and we must report the table is full.  If we do not, then we might
741     // end up deciding that the table is not full and inserting into the
742     // current table, while a 'get' has decided the same key cannot be in this
743     // table because of too many reprobes.  The invariant is:
744     //   slots.estimate_sum >= max_reprobe_cnt >= reprobe_limit(len)
745     private final boolean tableFull( int reprobe_cnt, int len ) {
746       return
747         // Do the cheap check first: we allow some number of reprobes always
748         reprobe_cnt >= REPROBE_LIMIT &&
749         // More expensive check: see if the table is > 1/4 full.
750         _slots.estimate_get() >= reprobe_limit(len);
751     }
752 
753     // --- resize ------------------------------------------------------------
754     // Resizing after too many probes.  "How Big???" heuristics are here.
755     // Callers will (not this routine) will 'help_copy' any in-progress copy.
756     // Since this routine has a fast cutout for copy-already-started, callers
757     // MUST 'help_copy' lest we have a path which forever runs through
758     // 'resize' only to discover a copy-in-progress which never progresses.
759     private final Object[] resize( NonBlockingIdentityHashMap topmap, Object[] kvs) {
760       assert chm(kvs) == this;
761 
762       // Check for resize already in progress, probably triggered by another thread
763       Object[] newkvs = _newkvs; // VOLATILE READ
764       if( newkvs != null )       // See if resize is already in progress
765         return newkvs;           // Use the new table already
766 
767       // No copy in-progress, so start one.  First up: compute new table size.
768       int oldlen = len(kvs);    // Old count of K,V pairs allowed
769       int sz = size();          // Get current table count of active K,V pairs
770       int newsz = sz;           // First size estimate
771 
772       // Heuristic to determine new size.  We expect plenty of dead-slots-with-keys
773       // and we need some decent padding to avoid endless reprobing.
774       if( sz >= (oldlen>>2) ) { // If we are >25% full of keys then...
775         newsz = oldlen<<1;      // Double size
776         if( sz >= (oldlen>>1) ) // If we are >50% full of keys then...
777           newsz = oldlen<<2;    // Double double size
778       }
779       // This heuristic in the next 2 lines leads to a much denser table
780       // with a higher reprobe rate
781       //if( sz >= (oldlen>>1) ) // If we are >50% full of keys then...
782       //  newsz = oldlen<<1;    // Double size
783 
784       // Last (re)size operation was very recent?  Then double again; slows
785       // down resize operations for tables subject to a high key churn rate.
786       long tm = System.currentTimeMillis();
787       long q=0;
788       if( newsz <= oldlen && // New table would shrink or hold steady?
789           tm <= topmap._last_resize_milli+10000 && // Recent resize (less than 1 sec ago)
790           (q=_slots.estimate_get()) >= (sz<<1) ) // 1/2 of keys are dead?
791         newsz = oldlen<<1;      // Double the existing size
792 
793       // Do not shrink, ever
794       if( newsz < oldlen ) newsz = oldlen;
795 
796       // Convert to power-of-2
797       int log2;
798       for( log2=MIN_SIZE_LOG; (1<<log2) < newsz; log2++ ) ; // Compute log2 of size
799 
800       // Now limit the number of threads actually allocating memory to a
801       // handful - lest we have 750 threads all trying to allocate a giant
802       // resized array.
803       long r = _resizers;
804       while( !_resizerUpdater.compareAndSet(this,r,r+1) )
805         r = _resizers;
806       // Size calculation: 2 words (K+V) per table entry, plus a handful.  We
807       // guess at 32-bit pointers; 64-bit pointers screws up the size calc by
808       // 2x but does not screw up the heuristic very much.
809       int megs = ((((1<<log2)<<1)+4)<<3/*word to bytes*/)>>20/*megs*/;
810       if( r >= 2 && megs > 0 ) { // Already 2 guys trying; wait and see
811         newkvs = _newkvs;        // Between dorking around, another thread did it
812         if( newkvs != null )     // See if resize is already in progress
813           return newkvs;         // Use the new table already
814         // TODO - use a wait with timeout, so we'll wakeup as soon as the new table
815         // is ready, or after the timeout in any case.
816         //synchronized( this ) { wait(8*megs); }         // Timeout - we always wakeup
817         // For now, sleep a tad and see if the 2 guys already trying to make
818         // the table actually get around to making it happen.
819         try { Thread.sleep(8*megs); } catch( Exception e ) { }
820       }
821       // Last check, since the 'new' below is expensive and there is a chance
822       // that another thread slipped in a new thread while we ran the heuristic.
823       newkvs = _newkvs;
824       if( newkvs != null )      // See if resize is already in progress
825         return newkvs;          // Use the new table already
826 
827       // Double size for K,V pairs, add 1 for CHM
828       newkvs = new Object[((1<<log2)<<1)+2]; // This can get expensive for big arrays
829       newkvs[0] = new CHM(_size); // CHM in slot 0
830       newkvs[1] = new int[1<<log2]; // hashes in slot 1
831 
832       // Another check after the slow allocation
833       if( _newkvs != null )     // See if resize is already in progress
834         return _newkvs;         // Use the new table already
835 
836       // The new table must be CAS'd in so only 1 winner amongst duplicate
837       // racing resizing threads.  Extra CHM's will be GC'd.
838       if( CAS_newkvs( newkvs ) ) { // NOW a resize-is-in-progress!
839         //notifyAll();            // Wake up any sleepers
840         //long nano = System.nanoTime();
841         //System.out.println(" "+nano+" Resize from "+oldlen+" to "+(1<<log2)+" and had "+(_resizers-1)+" extras" );
842         //if( System.out != null ) System.out.print("["+log2);
843         topmap.rehash();        // Call for Hashtable's benefit
844       } else                    // CAS failed?
845         newkvs = _newkvs;       // Reread new table
846       return newkvs;
847     }

849 
850     // The next part of the table to copy.  It monotonically transits from zero
851     // to _kvs.length.  Visitors to the table can claim 'work chunks' by
852     // CAS'ing this field up, then copying the indicated indices from the old
853     // table to the new table.  Workers are not required to finish any chunk;
854     // the counter simply wraps and work is copied duplicately until somebody
855     // somewhere completes the count.
856     volatile long _copyIdx = 0;
857     static private final AtomicLongFieldUpdater<CHM> _copyIdxUpdater =
858       AtomicLongFieldUpdater.newUpdater(CHM.class, "_copyIdx");
859 
860     // Work-done reporting.  Used to efficiently signal when we can move to
861     // the new table.  From 0 to len(oldkvs) refers to copying from the old
862     // table to the new.
863     volatile long _copyDone= 0;
864     static private final AtomicLongFieldUpdater<CHM> _copyDoneUpdater =
865       AtomicLongFieldUpdater.newUpdater(CHM.class, "_copyDone");
866 
867     // --- help_copy_impl ----------------------------------------------------
868     // Help along an existing resize operation.  We hope its the top-level
869     // copy (it was when we started) but this CHM might have been promoted out
870     // of the top position.
871     private final void help_copy_impl( NonBlockingIdentityHashMap topmap, Object[] oldkvs, boolean copy_all ) {
872       assert chm(oldkvs) == this;
873       Object[] newkvs = _newkvs;
874       assert newkvs != null;    // Already checked by caller
875       int oldlen = len(oldkvs); // Total amount to copy
876       final int MIN_COPY_WORK = Math.min(oldlen,1024); // Limit per-thread work
877 
878       // ---
879       int panic_start = -1;
880       int copyidx=-9999;            // Fool javac to think it's initialized
881       while( _copyDone < oldlen ) { // Still needing to copy?
882         // Carve out a chunk of work.  The counter wraps around so every
883         // thread eventually tries to copy every slot repeatedly.
884 
885         // We "panic" if we have tried TWICE to copy every slot - and it still
886         // has not happened.  i.e., twice some thread somewhere claimed they
887         // would copy 'slot X' (by bumping _copyIdx) but they never claimed to
888         // have finished (by bumping _copyDone).  Our choices become limited:
889         // we can wait for the work-claimers to finish (and become a blocking
890         // algorithm) or do the copy work ourselves.  Tiny tables with huge
891         // thread counts trying to copy the table often 'panic'.
892         if( panic_start == -1 ) { // No panic?
893           copyidx = (int)_copyIdx;
894           while( copyidx < (oldlen<<1) && // 'panic' check
895                  !_copyIdxUpdater.compareAndSet(this,copyidx,copyidx+MIN_COPY_WORK) )
896             copyidx = (int)_copyIdx;      // Re-read
897           if( !(copyidx < (oldlen<<1)) )  // Panic!
898             panic_start = copyidx;        // Record where we started to panic-copy
899         }
900 
901         // We now know what to copy.  Try to copy.
902         int workdone = 0;
903         for( int i=0; i<MIN_COPY_WORK; i++ )
904           if( copy_slot(topmap,(copyidx+i)&(oldlen-1),oldkvs,newkvs) ) // Made an oldtable slot go dead?
905             workdone++;         // Yes!
906         if( workdone > 0 )      // Report work-done occasionally
907           copy_check_and_promote( topmap, oldkvs, workdone );// See if we can promote
908         //for( int i=0; i<MIN_COPY_WORK; i++ )
909         //  if( copy_slot(topmap,(copyidx+i)&(oldlen-1),oldkvs,newkvs) ) // Made an oldtable slot go dead?
910         //    copy_check_and_promote( topmap, oldkvs, 1 );// See if we can promote
911 
912         copyidx += MIN_COPY_WORK;
913         // Uncomment these next 2 lines to turn on incremental table-copy.
914         // Otherwise this thread continues to copy until it is all done.
915         if( !copy_all && panic_start == -1 ) // No panic?
916           return;       // Then done copying after doing MIN_COPY_WORK
917       }
918       // Extra promotion check, in case another thread finished all copying
919       // then got stalled before promoting.
920       copy_check_and_promote( topmap, oldkvs, 0 );// See if we can promote
921     }

924     // --- copy_slot_and_check -----------------------------------------------
925     // Copy slot 'idx' from the old table to the new table.  If this thread
926     // confirmed the copy, update the counters and check for promotion.
927     //
928     // Returns the result of reading the volatile _newkvs, mostly as a
929     // convenience to callers.  We come here with 1-shot copy requests
930     // typically because the caller has found a Prime, and has not yet read
931     // the _newkvs volatile - which must have changed from null-to-not-null
932     // before any Prime appears.  So the caller needs to read the _newkvs
933     // field to retry his operation in the new table, but probably has not
934     // read it yet.
935     private final Object[] copy_slot_and_check( NonBlockingIdentityHashMap topmap, Object[] oldkvs, int idx, Object should_help ) {
936       assert chm(oldkvs) == this;
937       Object[] newkvs = _newkvs; // VOLATILE READ
938       // We're only here because the caller saw a Prime, which implies a
939       // table-copy is in progress.
940       assert newkvs != null;
941       if( copy_slot(topmap,idx,oldkvs,_newkvs) )   // Copy the desired slot
942         copy_check_and_promote(topmap, oldkvs, 1); // Record the slot copied
943       // Generically help along any copy (except if called recursively from a helper)
944       return (should_help == null) ? newkvs : topmap.help_copy(newkvs);
945     }

947     // --- copy_check_and_promote --------------------------------------------
948     private final void copy_check_and_promote( NonBlockingIdentityHashMap topmap, Object[] oldkvs, int workdone ) {
949       assert chm(oldkvs) == this;
950       int oldlen = len(oldkvs);
951       // We made a slot unusable and so did some of the needed copy work
952       long copyDone = _copyDone;
953       assert (copyDone+workdone) <= oldlen;
954       if( workdone > 0 ) {
955         while( !_copyDoneUpdater.compareAndSet(this,copyDone,copyDone+workdone) ) {
956           copyDone = _copyDone; // Reload, retry
957           assert (copyDone+workdone) <= oldlen;
958         }
959         //if( (10*copyDone/oldlen) != (10*(copyDone+workdone)/oldlen) )
960         //System.out.print(" "+(copyDone+workdone)*100/oldlen+"%"+"_"+(_copyIdx*100/oldlen)+"%");
961       }
962 
963       // Check for copy being ALL done, and promote.  Note that we might have
964       // nested in-progress copies and manage to finish a nested copy before
965       // finishing the top-level copy.  We only promote top-level copies.
966       if( copyDone+workdone == oldlen && // Ready to promote this table?
967           topmap._kvs == oldkvs && // Looking at the top-level table?
968           // Attempt to promote
969           topmap.CAS_kvs(oldkvs,_newkvs) ) {
970         topmap._last_resize_milli = System.currentTimeMillis(); // Record resize time for next check
971         //long nano = System.nanoTime();
972         //System.out.println(" "+nano+" Promote table to "+len(_newkvs));
973         //if( System.out != null ) System.out.print("]");
974       }
975     }

977     // --- copy_slot ---------------------------------------------------------
978     // Copy one K/V pair from oldkvs[i] to newkvs.  Returns true if we can
979     // confirm that the new table guaranteed has a value for this old-table
980     // slot.  We need an accurate confirmed-copy count so that we know when we
981     // can promote (if we promote the new table too soon, other threads may
982     // 'miss' on values not-yet-copied from the old table).  We don't allow
983     // any direct updates on the new table, unless they first happened to the
984     // old table - so that any transition in the new table from null to
985     // not-null must have been from a copy_slot (or other old-table overwrite)
986     // and not from a thread directly writing in the new table.  Thus we can
987     // count null-to-not-null transitions in the new table.
988     private boolean copy_slot( NonBlockingIdentityHashMap topmap, int idx, Object[] oldkvs, Object[] newkvs ) {
989       // Blindly set the key slot from null to TOMBSTONE, to eagerly stop
990       // fresh put's from inserting new values in the old table when the old
991       // table is mid-resize.  We don't need to act on the results here,
992       // because our correctness stems from box'ing the Value field.  Slamming
993       // the Key field is a minor speed optimization.
994       Object key;
995       while( (key=key(oldkvs,idx)) == null )
996         CAS_key(oldkvs,idx, null, TOMBSTONE);
997 
998       // ---
999       // Prevent new values from appearing in the old table.
1000       // Box what we see in the old table, to prevent further updates.
1001       Object oldval = val(oldkvs,idx); // Read OLD table
1002       while( !(oldval instanceof Prime) ) {
1003         final Prime box = (oldval == null || oldval == TOMBSTONE) ? TOMBPRIME : new Prime(oldval);
1004         if( CAS_val(oldkvs,idx,oldval,box) ) { // CAS down a box'd version of oldval
1005           // If we made the Value slot hold a TOMBPRIME, then we both
1006           // prevented further updates here but also the (absent)
1007           // oldval is vaccuously available in the new table.  We
1008           // return with true here: any thread looking for a value for
1009           // this key can correctly go straight to the new table and
1010           // skip looking in the old table.
1011           if( box == TOMBPRIME )
1012             return true;
1013           // Otherwise we boxed something, but it still needs to be
1014           // copied into the new table.
1015           oldval = box;         // Record updated oldval
1016           break;                // Break loop; oldval is now boxed by us
1017         }
1018         oldval = val(oldkvs,idx); // Else try, try again
1019       }
1020       if( oldval == TOMBPRIME ) return false; // Copy already complete here!
1021 
1022       // ---
1023       // Copy the value into the new table, but only if we overwrite a null.
1024       // If another value is already in the new table, then somebody else
1025       // wrote something there and that write is happens-after any value that
1026       // appears in the old table.  If putIfMatch does not find a null in the
1027       // new table - somebody else should have recorded the null-not_null
1028       // transition in this copy.
1029       Object old_unboxed = ((Prime)oldval)._V;
1030       assert old_unboxed != TOMBSTONE;
1031       boolean copied_into_new = (putIfMatch(topmap, newkvs, key, old_unboxed, null) == null);
1032 
1033       // ---
1034       // Finally, now that any old value is exposed in the new table, we can
1035       // forever hide the old-table value by slapping a TOMBPRIME down.  This
1036       // will stop other threads from uselessly attempting to copy this slot
1037       // (i.e., it's a speed optimization not a correctness issue).
1038       while( !CAS_val(oldkvs,idx,oldval,TOMBPRIME) )
1039         oldval = val(oldkvs,idx);
1040 
1041       return copied_into_new;
1042     } // end copy_slot
