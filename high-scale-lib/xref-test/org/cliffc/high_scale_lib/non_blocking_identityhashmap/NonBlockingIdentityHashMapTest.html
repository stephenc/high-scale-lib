26    // Throw a ClassCastException if I see a tombstone during key-compares
27    private static class KeyBonk {
28      final int _x;
29      KeyBonk( int i ) { _x=i; }
30      public boolean equals( Object o ) {
31        if( o == null ) return false;
32        return ((KeyBonk)o)._x    // Throw CCE here
33          == this._x; 
34      }
35      public int hashCode() { return (_x>>2); }
36      public String toString() { return "Bonk_"+Integer.toString(_x); }
37    }

310   static public void testNonBlockingIdentityHashMapIterator() throws InterruptedException {
311     final int ITEM_COUNT1 = 1000;
312     final int THREAD_COUNT = 5;
313     final int PER_CNT = ITEM_COUNT1/THREAD_COUNT;
314     final int ITEM_COUNT = PER_CNT*THREAD_COUNT; // fix roundoff for odd thread counts
315 
316     NonBlockingIdentityHashMap<Long,TestKey> nbhml = new NonBlockingIdentityHashMap<Long,TestKey>();
317     // use a barrier to open the gate for all threads at once to avoid rolling
318     // start and no actual concurrency
319     final CyclicBarrier barrier = new CyclicBarrier(THREAD_COUNT);
320     final ExecutorService ex = Executors.newFixedThreadPool(THREAD_COUNT);
321     final CompletionService<Object> co = new ExecutorCompletionService<Object>(ex);
322     for( int i=0; i<THREAD_COUNT; i++ ) {
323       co.submit(new NBHMLFeeder(nbhml, PER_CNT, barrier, i*PER_CNT));
324     }
325     for( int retCount = 0; retCount < THREAD_COUNT; retCount++ ) {
326       co.take();
327     }
328     ex.shutdown();
329 
330     assertEquals("values().size()", ITEM_COUNT, nbhml.values().size());
331     assertEquals("entrySet().size()", ITEM_COUNT, nbhml.entrySet().size());
332     int itemCount = 0;
333     for( TestKey K : nbhml.values() )
334       itemCount++;
335     assertEquals("values().iterator() count", ITEM_COUNT, itemCount);
336   }

341   static private class NBHMLFeeder implements Callable<Object> {
342     static private final Random _rand = new Random(System.currentTimeMillis());
343     private final NonBlockingIdentityHashMap<Long,TestKey> _map;
344     private final int _count;
345     private final CyclicBarrier _barrier;
346     private final long _offset;
347     public NBHMLFeeder(final NonBlockingIdentityHashMap<Long,TestKey> map, final int count, final CyclicBarrier barrier, final long offset) {
348       _map = map;
349       _count = count;
350       _barrier = barrier;
351       _offset = offset;
352     }
353     public Object call() throws Exception {
354       _barrier.await();         // barrier, to force racing start
355       for( long j=0; j<_count; j++ )
356         _map.put(j+_offset, new TestKey(_rand.nextLong(),_rand.nextInt (), (short) _rand.nextInt(Short.MAX_VALUE)));
357       return null;
358     }

432     private final Hashtable<Integer, List<TestKey>> _items = new Hashtable<Integer, List<TestKey>>();
433     private int _size = 0;
434     public int size() { return _size;  }
435     // Put items into the hashtable, sorted by 'type' into LinkedLists.
436     public void checkedPut(final long id, final int type, final int hash) {
437       _size++;
438       final TestKey item = new TestKey(id, type, hash);
439       if( !_items.containsKey(type) )
440         _items.put(type, new LinkedList<TestKey>());
441       _items.get(type).add(item);
442     }

444     public NonBlockingIdentityHashMap<Long,TestKey> getMapMultithreaded() throws InterruptedException, ExecutionException {
445       final int threadCount = _items.keySet().size();
446       final NonBlockingIdentityHashMap<Long,TestKey> map = new NonBlockingIdentityHashMap<Long,TestKey>();
447 
448       // use a barrier to open the gate for all threads at once to avoid rolling start and no actual concurrency
449       final CyclicBarrier barrier = new CyclicBarrier(threadCount);
450       final ExecutorService ex = Executors.newFixedThreadPool(threadCount);
451       final CompletionService<Integer> co = new ExecutorCompletionService<Integer>(ex);
452       for( Integer type : _items.keySet() ) {
453         // A linked-list of things to insert
454         List<TestKey> items = _items.get(type);
455         TestKeyFeederThread feeder = new TestKeyFeederThread(type, items, map, barrier);
456         co.submit(feeder);
457       }
458 
459       // wait for all threads to return
460       int itemCount = 0;
461       for( int retCount = 0; retCount < threadCount; retCount++ ) {
462         final Future<Integer> result = co.take();
463         itemCount += result.get();
464       }
465       ex.shutdown();
466       return map;
467     }
468   }

472     private final int _type;
473     private final NonBlockingIdentityHashMap<Long,TestKey> _map;
474     private final List<TestKey> _items;
475     private final CyclicBarrier _barrier;
476     public TestKeyFeederThread(final int type, final List<TestKey> items, final NonBlockingIdentityHashMap<Long,TestKey> map, final CyclicBarrier barrier) {
477       _type = type;
478       _map = map;
479       _items = items;
480       _barrier = barrier;
481     }

483     public Integer call() throws Exception {
484       _barrier.await();
485       int count = 0;
486       for( TestKey item : _items ) {
487         if (_map.contains(item._id)) {
488           System.err.printf("COLLISION DETECTED: %s exists\n", item.toString());
489         }
490         final TestKey exists = _map.putIfAbsent(item._id, item);
491         if (exists == null) {
492           count++;
493         } else {
494           System.err.printf("COLLISION DETECTED: %s exists as %s\n", item.toString(), exists.toString());
495         }
496       }
497       return count;
498     }
